%% Template: ats-flex-resume
%% A clean, single-column, icon-free, table-free U.S.-style resume for
%% business operations / marketing analyst / project coordinator / CRM /
%% reporting roles. Fits 1 page by default; flows to 2 pages only if the
%% content genuinely requires it (no forced page count, no font-shrinking
%% hacks to force-fit).
%%
%% Compile with: lualatex (also works with xelatex unchanged -- no
%% fontspec/custom fonts, so there is nothing engine-specific here beyond
%% the two both being fontspec-capable if you later add a custom font)
%% Compile command: lualatex -interaction=nonstopmode <file>.tex

\documentclass[11pt,letterpaper]{article}

\usepackage[margin=0.65in]{geometry}
\usepackage[T1]{fontenc}
\usepackage{lmodern}
\renewcommand{\familydefault}{\sfdefault}
\usepackage[hidelinks]{hyperref}
\usepackage{titlesec}
\usepackage{enumitem}

% Fully monochrome by design -- no xcolor, no accent color. Maximizes
% compatibility with ATS parsers/reviewers that flag colored text, and
% keeps the template usable with zero changes under any TeX engine.
\titleformat{\section}{\normalsize\bfseries\scshape}{}{0em}{}[\titlerule]
\titlespacing{\section}{0pt}{10pt}{4pt}

\setlist[itemize]{leftmargin=1.1em, itemsep=1pt, topsep=1pt, parsep=0pt, label=\textbullet}

\setlength{\parindent}{0pt}
\pagenumbering{gobble}

% ------------------------------------------------------------------
% \cvheader{Title}{Company, Location}{Dates}
% Plain-text two-line block -- no tabular/tabularx anywhere in this
% template. Alignment is done with \hfill on a single \noindent line,
% which extracts as ordinary left-to-right text (a table would risk an
% ATS parser reading cells out of order).
% ------------------------------------------------------------------
\newcommand{\cvheader}[3]{%
  \noindent\textbf{#1}\hfill{#3}\\
  \textit{#2}
}
